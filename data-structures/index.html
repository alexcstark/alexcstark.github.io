I've spent some time coding out classic data structures.

Linked List

class Node {  
  constructor(data, next){
    this.data = data;
    this.next = next || null;
  }
}
class LinkedList {  
  constructor(){
    this.head = null;
    this.length = 0;
  }

  add(data){
    var node = new Node(data);
    var current = this.head;
    if(this.head !== null){
      while(current.next){
        current = current.next;
      }
      current.next = node;
    } else {
      this.head = node;
    }
    this.length++;
    return this.head;
  }

  remove(index){
    if(index < 0 || index > this.length) return "ERROR";

    if(this.head === null) return "Nothing to remove";
    if(index === 0){
      this.head = this.head.next;
    } else {
      var current = this.head;
      while(index > 1){
        current = current.next;
        index--;
      }
      current.next = current.next.next
      return this.head
    }
  }
}

var newList = new LinkedList();  
newList.add(1);  
newList.add(2);  
newList.add(3);  
newList.remove(0);  
console.log(newList.head);


function generateLinkedLists(arrayofData){  
  var nodeStorage = {};
  arrayofData.forEach((data, i) => {
    nodeStorage['node' + i] = new Node(data);
    if(i !== 0){
      nodeStorage['node' + (i - 1)].next = nodeStorage['node' + i]
    }
  });
  return nodeStorage['node0'];
}

Doubly Linked List

class Node {  
  constructor(data, next, prev){
    this.data = data;
    this.next = next || null;
    this.prev = prev || null;
  }
}
class DoublyLinkedList {  
  constructor(){
    this.head = null;
    this.tail = null
    this.length = 0;
  }

  add(data){
    var node = new Node(data);
    var current = this.head;
    if(this.head !== null){
      while(current.next){
        current = current.next;
      }
      current.next = node;
      node.prev = current;
      this.tail = node;
    } else {
      this.head = node;
    }
    this.length++;
    return this.head;
  }

  remove(index){
    if(index < 0 || index > this.length) return "ERROR";

    if(this.head === null) return "Nothing to remove";
    if(index === 0){
      this.head = this.head.next;
      this.head.prev = null;
    } else {
      var current = this.head;
      while(index > 1){
        current = current.next;
        index--;
      }
      current.next = current.next.next;
      current.next.next.prev = current.next;
      return this.head
    }
  }
}

var newList = new DoublyLinkedList();  
newList.add(1);  
newList.add(2);  
newList.add(3);  
newList.remove(0);  
console.log(newList.head);
